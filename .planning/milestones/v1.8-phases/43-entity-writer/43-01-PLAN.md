   ```python
   _DEP_NAME_RE = re.compile(r"^[A-Za-z0-9_.\-]+")
   def _extract_dep_name(pep508_str: str) -> str | None:
       """Extract the bare package name from a PEP 508 specifier.

       Returns lowercase name, or None if the string doesn't begin with a
       valid identifier. Strips bracketed extras (`[bedrock]`), version
       specifiers, environment markers, and URL forms (`git+...#egg=...`

       s = pep508_str.strip()
       if not s or s.startswith(("git+", "http://", "https://", "-e ", ".")):
           return None
       m = _DEP_NAME_RE.match(s)
       return m.group(0).lower() if m else None
   ```

   ```python
       import_root: Path, skip_dirs: frozenset[str], repo_root: Path
   ) -> Iterator[Path]:
       """Yield each __init__.py-containing subdirectory STRICTLY UNDER `import_root`
       (excludes `import_root` itself — D-43-01).

       Honors skip_dirs via _ignore.should_skip. Does not follow symlinks

       if not (import_root / "__init__.py").exists():

       for dirpath, dirnames, filenames in os.walk(import_root, followlinks=False):
           d = Path(dirpath)
           # Skip the import root itself (fix for folded todo).
           if d.resolve() == import_root_resolved:

               dirnames[:] = [

                   if not _ignore.should_skip(name, skip_dirs)
   ```
