3. Add the table builder. After Plan 01's `rewrite_text` function:

   ```python
   import sqlite3
   from typing import Callable, Iterable

   from graph_io import queries as _queries
   from wiki_io.entity_writer import encode_slug as _encode_slug


   _LIST_FNS: dict[str, Callable[[sqlite3.Connection], Iterable]] = {
       "package":    _queries.list_packages,
       "dependency": _queries.list_dependencies,
       "domain":     _queries.list_domains,
       "plugin":     _queries.list_plugins,
       "test_suite": _queries.list_test_suites,
   }


   def _new_slug(uri: str) -> str:
       return f"entities/{_encode_slug(uri)}"


   def _build_source1_and_index(conn: sqlite3.Connection) -> tuple[dict[str, str | None], dict[str, dict[str, object]]]:
       """Source 1 + in-memory (kind, name[, ecosystem]) index for Source 2/3 lookups.

       Returns (table_partial, index) where:
         table_partial: bare + wiki/-prefixed entries from convention templates
         index: {kind: {key: new_slug}} where key is `name` (or `f"{ecosystem}/{name}"` for dependency)
       """
       table: dict[str, str | None] = {}
       index: dict[str, dict[str, object]] = {k: {} for k in CONVENTION_TEMPLATES}
       for kind, template in CONVENTION_TEMPLATES.items():
           list_fn = _LIST_FNS[kind]
           for node in list_fn(conn):
               uri = node.attrs.get("uri") if isinstance(node.attrs, dict) else None
               if not uri:
                   continue
               new_slug = _new_slug(uri)
               name = node.name
               if kind == "dependency":
                   ecosystem = node.attrs.get("ecosystem", "")
                   if not ecosystem:
                       continue  # defensive — deps always have ecosystem in v1.7+
                   bare = template.format(name=name, ecosystem=ecosystem)
                   index[kind][f"{ecosystem}/{name}"] = new_slug
               else:
                   bare = template.format(name=name)
                   index[kind][name] = new_slug
               table[bare] = new_slug
               table[f"wiki/{bare}"] = new_slug
       return table, index


   def _source2_scan_old_layout(wiki_root: Path, table: dict[str, str | None], index: dict[str, dict[str, object]]) -> None:
       """Source 2: scan-and-match over old layout dirs. Mutates `table` in place.

       For each .md file under wiki/<root>/ where root is in OLD_LAYOUT_ROOTS,
       derive the entity name from the path, look up via `index`, and add an
       entry if matched. package-family files are skipped (no entries added) per
       CONTEXT D-04.
       """
       kind_for_root = {
           "packages": "package",
           "dependencies": "dependency",
           "domain": "domain",
           "plugin": "plugin",
           "package-family": None,  # deferred per D-04
       }
       for root_name in OLD_LAYOUT_ROOTS:
           root = wiki_root / root_name
           if not root.is_dir():
               continue
           kind = kind_for_root.get(root_name)
           for md in root.rglob("*.md"):
               if kind is None:
                   continue  # package-family — leave unmatched per D-04
               # Compute the old_target (path-without-.md, relative to wiki_root).
               try:
                   old_target = str(md.relative_to(wiki_root).with_suffix("")).replace(os.sep, "/")
               except ValueError:
                   continue
               # Skip if already in table (Source 1 covered it).
               if old_target in table or f"wiki/{old_target}" in table:
                   continue
               # Derive name from path.
               parts = md.relative_to(root).parts
               if kind == "dependency":
                   # path shape: <ecosystem>/<name>/overview.md
                   if len(parts) < 2:
                       continue
                   key = f"{parts[0]}/{parts[1]}"  # ecosystem/name
               else:
                   # path shape: <name>/index.md  (or similar)
                   if len(parts) < 1:
                       continue
                   key = parts[0]
               new_slug = index.get(kind, {}).get(key)
               if new_slug is None:
                   continue  # not in graph; leave uncovered (Source 3 may catch it)
               table[old_target] = new_slug
               table[f"wiki/{old_target}"] = new_slug


   def _source3_grep_curated_lanes(wiki_root: Path, table: dict[str, str | None], index: dict[str, dict[str, object]], log_path: Path | None) -> None:
       """Source 3: grep curated lanes for inbound links to old-layout prefixes.

       For each .md file under each curated lane, find wikilinks in non-code
       regions whose target starts with an OLD_LAYOUT_PREFIXES entry. If the
       target is not already in `table`, attempt to derive the entity name from
       the path components and look up via `index`. If matched, add. If not,
       add as None and log to migration.log.
       """
       workspace_root = wiki_root.parent
       lanes = [
           wiki_root / "concepts",
           wiki_root / "adrs",
           wiki_root / "architecture",
           wiki_root / "sources",
           workspace_root / "work",
       ]
       kind_for_prefix = {
           "packages/": "package",
           "wiki/packages/": "package",
           "dependencies/": "dependency",
           "wiki/dependencies/": "dependency",
           "domain/": "domain",
           "wiki/domain/": "domain",
           "plugin/": "plugin",
           "wiki/plugin/": "plugin",
           "test-suites/": "test_suite",
           "wiki/test-suites/": "test_suite",
           "package-family/": None,
           "wiki/package-family/": None,
       }

                       continue

                       _record_unresolvable(table, log_path, md, workspace_root, target)
                       continue

                   tail = target[len(matched_prefix):]
                   tail_parts = tail.split("/")
                   if kind == "dependency":

                           _record_unresolvable(table, log_path, md, workspace_root, target)
                           continue
                       key = f"{tail_parts[0]}/{tail_parts[1]}"
                   else:
                       if not tail_parts:
                           _record_unresolvable(table, log_path, md, workspace_root, target)
                           continue
                       key = tail_parts[0]
                   new_slug = index.get(kind, {}).get(key)
                   if new_slug is None:
                       _record_unresolvable(table, log_path, md, workspace_root, target)
                       continue
                   table[target] = new_slug

       """Mark `target` as unresolvable in the table and log a JSONL line."""

               "phase": "unresolved",

       Source 2: scan-and-match over wiki/{packages,dependencies,domain,plugin,package-family}/.
       Source 3: grep the 5 curated lanes for inbound old-layout wikilinks; log unresolvables.

       table, index = _build_source1_and_index(conn)

       """Write `content` to `path` atomically (temp file + os.replace)."""

       ts = _utc_iso_z()
       for m in WIKILINK_RE.finditer(text_before):

               continue
           target = m.group(1).strip()
           new_slug = table.get(target)
           if new_slug is None:
               continue

               "phase": "rewrite",
               "file": file_rel,
               "from": target,
               "to": new_slug,
           })

       wiki_root: Path,
       table: dict[str, str | None],
       *,
       log_path: Path | None = None,
       lanes: list[Path] | None = None,
   ) -> RewriteResult:
       """Walk the 5 curated lanes and rewrite wikilinks per the table.

       Per CONTEXT D-13: defaults walk `wiki/concepts`, `wiki/adrs`,
       `wiki/architecture`, `wiki/sources`, and workspace-rooted `work/`
       (sibling of wiki/). Per D-14: `wiki/` root files are never visited.

       For each `.md` file in scope, call `rewrite_text(text, table)`; if
       count > 0, write back atomically (temp-file + os.replace) and append
       one JSONL line per rewrite to `log_path` (if not None).

       Returns RewriteResult with totals + per_file counts.
       """
       workspace_root = wiki_root.parent
       lanes_resolved = lanes if lanes is not None else _default_lanes(wiki_root)
       files_scanned = 0
       files_modified = 0
       rewrites_total = 0
       unresolved_total = 0
       per_file: dict[str, int] = {}
       for lane in lanes_resolved:
           if not lane.is_dir():
               continue
           for md in sorted(lane.rglob("*.md")):
               files_scanned += 1
               try:
                   text = md.read_text(encoding="utf-8")
               except OSError:
                   continue
               unresolved_total += _count_unresolved_in_text(text, table)
               new_text, count = rewrite_text(text, table)
               if count > 0:
                   _atomic_write_text(md, new_text)
                   files_modified += 1
                   rewrites_total += count
                   try:
                       rel = str(md.relative_to(workspace_root))
                   except ValueError:
                       rel = str(md)
                   per_file[rel] = count
                   if log_path is not None:
                       _log_rewrites(text, new_text, table, log_path, rel)
       return RewriteResult(
           files_scanned=files_scanned,
           files_modified=files_modified,
           rewrites_total=rewrites_total,
           unresolved_total=unresolved_total,
           per_file=per_file,
       )
   ```
