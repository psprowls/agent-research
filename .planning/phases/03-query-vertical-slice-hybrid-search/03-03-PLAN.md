- @dataclass class QueryResult:
      answer: str
      citations: list[str]
      pages_drilled: int
      search_scores: dict[str, dict[str, float]]
        # {page_path: {"bm25": float, "embed": float, "rrf": float}}

- LIBRARIAN_SYSTEM: str   — module-level constant per AI-SPEC §4b.3
- SYNTHESIZER_SYSTEM: str — module-level constant
  Use the exact strings from AI-SPEC §3 "Entry Point Pattern" lines 216-228.

- async def run_query(
      query: str,
      vault_path: Path | None = None,
      top_k: int = 5,
  ) -> QueryResult
  Steps:
    1. wiki, _ = resolve_wiki_and_repo(vault_path)
    2. Auto-build: if not (wiki/.code-wiki/bm25).exists() or
                   not (wiki/.code-wiki/search.db).exists():
         logger.warning("First-time index build — may take a moment.")
         build_index(wiki)
    3. BM25 search via bm25_query(query, wiki, top_k * 3) — over-retrieve per AI-SPEC §4b.4
    4. Embedding search:
         embeddings = BedrockEmbeddings(
             model_id="amazon.titan-embed-text-v2:0",
             region_name="us-east-1", normalize=True,
         )
         query_vec = embeddings.embed_query(query)
         embed_hits = _cosine_search_sqlite(wiki, query_vec, top_k * 3)
    5. RRF fusion (top_k * 3 each side -> top_k after fuse):
         bm25_rank_map = {path: i+1 for i, path in enumerate(bm25_paths)}
         embed_rank_map = {path: i+1 for i, (path, _) in enumerate(embed_hits)}
         fused = _rrf_fuse(bm25_rank_map, embed_rank_map)
         top_pages = sorted(fused, key=fused.get, reverse=True)[:top_k]
    6. Librarian fan-out:
         lib_cfg = load_role_config("librarian")
         librarian_llm = make_llm("librarian")
         pool = SubagentPool(trace_dir=wiki / ".code-wiki" / "traces")

         async def drill_page(page_path: str) -> str:
             page_text = (wiki / page_path).read_text()
             # Truncation guard per AI-SPEC §4b.4 + §4 Context Window Strategy:
             if len(page_text) > 24000:
                 page_text = page_text[:24000] + "\n[TRUNCATED]"
                 logger.warning("Truncated oversized page: %s", page_path)
             msgs = [
                 SystemMessage(content=LIBRARIAN_SYSTEM),
                 HumanMessage(content=f"Query: {query}\n\nPage ({page_path}):\n{page_text}"),
             ]
             resp = await librarian_llm.ainvoke(msgs)
             return resp.content

         fan_result = await pool.run_all(
             items=top_pages,
             task=drill_page,
             role="librarian",
             model_id=lib_cfg["model_id"],
             max_concurrency=lib_cfg["max_concurrency"],
         )
    7. Synthesizer call (single, awaited):
         excerpts_text = "\n\n---\n\n".join(
             f"[{item}]\n{result}"
             for item, result in fan_result.successes
             if (result or "").strip() != "NO_RELEVANT_CONTENT"
         )
         # Optional safety truncation per AI-SPEC §4b.4:
         if len(excerpts_text) > 60000:
             logger.warning("Truncating librarian excerpts before synthesis")
             excerpts_text = excerpts_text[:60000]
         synth_llm = make_llm("synthesizer")
         synth_msgs = [
             SystemMessage(content=SYNTHESIZER_SYSTEM),
             HumanMessage(content=f"Query: {query}\n\nLibrarian excerpts:\n{excerpts_text}"),
         ]
         synth_resp = await synth_llm.ainvoke(synth_msgs)
    8. Build QueryResult with:
         answer = synth_resp.content (a string)
         citations = _extract_wikilinks(synth_resp.content)
         pages_drilled = len(fan_result.successes)
         search_scores per top_page: bm25/embed/rrf raw scores
    9. Apply guardrails (G1, G4) via apply_guardrails(result, wiki, fan_result).
   10. Return guarded QueryResult.

- def _extract_wikilinks(text: str) -> list[str]:
      return re.findall(r"\[\[([^\]]+)\]\]", text)

- def apply_guardrails(result: QueryResult, vault_path: Path,
                      fan_result: FanOutResult) -> QueryResult:
  G1: For each [[wikilink]] in result.answer, check if vault_path / f"{link}.md"
      exists (case-insensitive via `list(vault_path.glob(f"**/{link}.md"))`).
      Unresolved → append `[warning: N citation(s) did not resolve: <list>]`
      to result.answer. Do NOT clear citations.
  G4: If not fan_result.successes (empty) AND result.citations non-empty:
      Prepend `[warning: no librarian excerpts; answer is unsupported by retrieved pages]`
      to result.answer, AND clear result.citations to [] to avoid G1 false-positives.
      Return mutated copy.

CLI subcommand signature (cli.py, follows existing trace pattern):
- @app.command()
  def query(
      query_text: str = typer.Argument(..., help="Natural language query"),
      top_k: int = typer.Option(5, "--top-k", help="Pages to drill (3-10)", min=3, max=10),
      vault: str = typer.Option("", "--vault", help="Vault path (default: env var)"),
      json_output: bool = typer.Option(False, "--json", help="Emit QueryResult as JSON"),
      no_state_gate: bool = typer.Option(False, "--no-state-gate", help="No-op; query is read-only"),
      quiet: bool = typer.Option(False, "--quiet", help="Suppress progress output (headless mode)"),
  ) -> None:
  Behavior:
    - vault_path = Path(vault) if vault else None
    - try: result = asyncio.run(run_query(query_text, vault_path, top_k=top_k))
      except RuntimeError as e:
          typer.echo(f"Error: {e}", err=True); raise typer.Exit(code=1)
    - If json_output: typer.echo(json.dumps(dataclasses.asdict(result), indent=2))
    - Else: typer.echo(result.answer); if citations: typer.echo(f"\nCitations: {', '.join(citations)}"); if not quiet: typer.echo(f"Pages drilled: {result.pages_drilled}", err=not sys.stdout.isatty())
    - Exit code 3 if result.pages_drilled < top_k (partial success per CLI-06)
