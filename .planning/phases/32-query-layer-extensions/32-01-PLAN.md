      def find(
          conn: sqlite3.Connection,
          *,
          name: str | None = None,
          kind: str | None = None,
      ) -> list[NodeRecord]:
          if kind is not None and kind not in _VALID_KINDS:
              raise ValueError(
                  f"unknown kind {kind!r}; valid: {sorted(_VALID_KINDS)}"
              )
          if name is None and kind is None:
              raise ValueError("find requires at least one of name or kind")
          if name is not None and kind is None:
              rows = conn.execute(
                  "SELECT kind, name, path, line, attrs_json "
                  "FROM nodes WHERE name = ?",
                  (name,),
              ).fetchall()
          elif name is not None and kind is not None:
              rows = conn.execute(
                  "SELECT kind, name, path, line, attrs_json "
                  "FROM nodes WHERE name = ? AND kind = ?",
                  (name, kind),
              ).fetchall()
          else:  # name is None and kind is not None
              rows = conn.execute(
                  "SELECT kind, name, path, line, attrs_json "
                  "FROM nodes WHERE kind = ? ORDER BY name",
                  (kind,),
              ).fetchall()
          return [_row_to_node(r) for r in rows]
