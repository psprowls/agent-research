```python
"""End-to-end regression: bootstrap a wiki + render package/app/plugin overview
templates + lint, asserting zero broken wikilinks across all three container
types. This is the HYGIENE-14 closure artifact per Phase 35 D-02/D-03.

The roadmap originally called for a manual /graph-wiki:query smoke transcript;
D-03 supersedes that with this automated test — a regression test that runs on
every CI is strictly stronger evidence than a one-time manual capture.
"""

from __future__ import annotations

import datetime as dt
from pathlib import Path

import pytest


def test_bootstrap_then_scan_then_lint_zero_broken_links(tmp_path: Path, monkeypatch) -> None:
    """Bootstrap a wiki + render the three overview templates + lint.

    Covers: HYGIENE-01 (wiki/ prefix), HYGIENE-02 (section index stubs),
    HYGIENE-03 (CONTAINER_DIR substitution for non-`packages/`),
    HYGIENE-06 (overview.md naming).
    """
    from wiki_io.init_vault import init_wiki, render_template

    # --- Build a minimal repo fixture covering all three container types ---
    repo = tmp_path / "repo"
    (repo / "packages" / "test-pkg" / "src" / "test_pkg").mkdir(parents=True)
    (repo / "packages" / "test-pkg" / "pyproject.toml").write_text(
        '[project]\nname = "test-pkg"\nversion = "0.1.0"\n', encoding="utf-8"
    )
    (repo / "agents" / "test-agent" / "src" / "test_agent").mkdir(parents=True)
    (repo / "agents" / "test-agent" / "pyproject.toml").write_text(
        '[project]\nname = "test-agent"\nversion = "0.1.0"\n', encoding="utf-8"
    )
    (repo / "plugins" / "test-plugin").mkdir(parents=True)
    (repo / "plugins" / "test-plugin" / "pyproject.toml").write_text(
        '[project]\nname = "test-plugin"\nversion = "0.1.0"\n', encoding="utf-8"
    )
    # Workspace-root pyproject so resolve_wiki_and_repo can find the repo:
    (repo / "pyproject.toml").write_text('[project]\nname = "tmp-repo"\nversion = "0.0.0"\n', encoding="utf-8")
    # Initialize as a git repo so workspace-io's _is_inside_git_repo returns True:
    import subprocess
    subprocess.run(["git", "init", "-q", str(repo)], check=True)

    # --- Run init_wiki against tmp_path/workspace ---
    workspace = tmp_path / "workspace"
    workspace.mkdir()
    monkeypatch.setenv("GRAPH_WIKI_WORKSPACE", str(workspace))

    # Pre-seed a minimal .graph-wiki.yaml so workspace-io's resolve does not error
    # (mirrors what `graph-wiki-agent bootstrap` would create via _ws_init):
    (workspace / ".graph-wiki.yaml").write_text(
        f"version: 2\nrepo-directory: {repo}\nplugins: []\n", encoding="utf-8"
    )

    wiki = workspace / "wiki"
    wiki.mkdir()
    result = init_wiki(
        wiki_path=wiki,
        repo_path=repo,
        topic="test",
        tool="claude-code",
        force=False,
        non_interactive=True,
        as_json=False,
    )

    # --- Render the three overview templates into wiki/ ---
    assets = Path(__file__).resolve().parents[1] / "src" / "wiki_io" / "assets" / "page-templates"
    today = dt.date.today().isoformat()

    # Helper: render an overview + minimal sub-page stubs so the wikilinks from the
    # overview resolve via link_targets (see lint_wiki.scan()'s link resolution).
    def render_container(container: str, slug: str, title_var: str, slug_var: str, tmpl_subdir: str) -> None:
        target_dir = wiki / container / slug
        target_dir.mkdir(parents=True, exist_ok=True)
        # Overview
        render_template(
            assets / tmpl_subdir / "overview.md",
            target_dir / "overview.md",
            {title_var: slug, slug_var: slug, "CONTAINER_DIR": container, "DATE": today},
        )
        # Sub-pages — write minimal stubs so wikilink targets resolve.
        # Match the sub-page set referenced by each template:
        #   package: api, patterns, work, testing, context
        #   app:     testing (only sub-page referenced in app/overview.md)
        #   plugin:  api, patterns, work, testing, context
        if tmpl_subdir in ("package", "plugin"):
            for sub in ("api", "patterns", "work", "testing", "context"):
                (target_dir / f"{sub}.md").write_text(
                    f"---\ncategory: {tmpl_subdir}\nstatus: active\n---\n# {slug}/{sub} stub\n",
                    encoding="utf-8",
                )
        elif tmpl_subdir == "app":
            (target_dir / "testing.md").write_text(
                f"---\ncategory: app\nstatus: active\n---\n# {slug}/testing stub\n",
                encoding="utf-8",
            )

    render_container("packages", "test-pkg", "PACKAGE_TITLE", "PACKAGE_SLUG", "package")
    render_container("apps", "test-app", "APP_TITLE", "APP_SLUG", "app")
    render_container("plugins", "test-plugin", "PACKAGE_TITLE", "PACKAGE_SLUG", "plugin")

    # --- Lint the populated wiki ---
    from wiki_io.lint_wiki import scan as lint_scan
    report = lint_scan(wiki, stale_days=90, log_gap_days=14, repo_path=None)

    # --- Assert zero broken wikilinks across all three container types ---
    # NOTE: read the report shape directly from lint_wiki.scan()'s return.
    # The broken-wikilink data is exposed via the key(s) that the existing
    # test_lint_wiki.py asserts against. Identify and pin the assertion to that
    # exact key (or its summary count). For example:
    #   assert report["broken_wikilinks"] == []
    # OR if the shape groups by page:
    #   total_broken = sum(len(v) for v in report["broken_wikilinks_by_page"].values())
    #   assert total_broken == 0
    # The exact assertion is determined by reading scan()'s return statement
    # — record the key name used in the SUMMARY for traceability.
    # If for any reason scan() returns broken_wikilinks > 0 in this fixture,
    # the test must fail loudly (not silently skip).
    broken = _extract_broken_wikilinks(report)
    assert broken == [], (
        f"Bootstrap-and-lint regression: expected zero broken wikilinks across "
        f"package/app/plugin containers, got: {broken!r}\n"
        f"Full report: {report!r}"
    )


def _extract_broken_wikilinks(report) -> list:
    """Return a flat list of broken-wikilink entries from a lint_wiki.scan report.

    Adapts to whichever key shape scan() uses (flat list vs per-page dict).
    """
    if "broken_wikilinks" in report:
        return list(report["broken_wikilinks"])
    if "broken_links" in report:
        # Some scan variants flatten into broken_links — handle gracefully.
        return list(report["broken_links"])
    # Per-page dict shape
    if "broken_wikilinks_by_page" in report:
        flat = []
        for page, links in report["broken_wikilinks_by_page"].items():
            flat.extend((page, link) for link in links)
        return flat
    # If scan() does not expose broken-wikilink data under any of these keys,
    # fail the test loudly so the executor reads the actual shape and pins it.
    raise AssertionError(
        f"lint_wiki.scan() returned a report without a recognized broken-wikilink "
        f"key — read scan()'s return statement and update _extract_broken_wikilinks "
        f"to match. report keys: {sorted(report.keys()) if hasattr(report, 'keys') else report!r}"
    )
```

1. **Read `lint_wiki.scan()`'s return statement FIRST** — the exact key for broken-wikilink output is the production datum that pins the assertion shape. Do not guess. Common shapes:
   - Flat list under `"broken_wikilinks"` or `"broken_links"`.
   - Per-page dict under `"broken_wikilinks_by_page"`.
   - Or a derived `total_broken_links` integer summary.

   If the actual shape is none of the above, surface it in `_extract_broken_wikilinks` so the test fails loudly with a message pointing the executor to the right code path.

2. **Sub-page stubs must satisfy `link_targets`.** The overview template emits `[[wiki/<container>/<slug>/api|api]]` etc. For lint to recognize those as resolved, sibling files at `<container>/<slug>/api.md`, `patterns.md`, etc. must exist (so `link_targets.add("wiki/<container>/<slug>/api")` fires in `lint_wiki.scan()` first loop). The minimal stub content (`# <slug>/<sub> stub`) is enough; lint only cares about presence, not content.

3. **`init_wiki()` creates the section index stubs** (`concepts/index.md`, etc.) via `SECTION_INDEX_STUBS`. That handles HYGIENE-02 coverage automatically — no additional work in the test.
