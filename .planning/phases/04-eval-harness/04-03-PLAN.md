deepeval 4.0 API:
```python
from deepeval.metrics import GEval
from deepeval.models import AmazonBedrockModel
from deepeval.test_case import LLMTestCase, SingleTurnParams

judge = AmazonBedrockModel(
    model="us.anthropic.claude-sonnet-4-6",  # MUST be explicit
    region="us-east-1",
    cost_per_input_token=3.0 / 1_000_000,   # per-token (not per-1M)
    cost_per_output_token=15.0 / 1_000_000,
    generation_kwargs={"temperature": 0},
)

metric = GEval(
    name="wiki_query_quality",
    criteria="Assess quality...",
    evaluation_steps=["step 1", ...],
    evaluation_params=[SingleTurnParams.INPUT, SingleTurnParams.ACTUAL_OUTPUT, SingleTurnParams.EXPECTED_OUTPUT],
    model=judge,   # NEVER omit — deepeval defaults to OpenAI GPT if missing
    threshold=0.5,
)
metric.measure(test_case)
# metric.score -> float 0.0-1.0
# metric.reason -> str
```

pytest-evals 0.3.4 API:
```python
@pytest.mark.eval(name="query_sweep")
@pytest.mark.parametrize("case", CASES)
async def test_case(case, eval_bag):
    eval_bag.score = 0.7  # store per-case metrics

@pytest.mark.eval_analysis(name="query_sweep")
def test_analysis(eval_results):
    scores = [r.score for r in eval_results]
    assert sum(scores)/len(scores) >= 0.5
```
