              return [], []

          def _resolve_path(value: str) -> str | None:
              """Resolve a package.json relative path to a repo-relative File path."""
              if "*" in value:
                  return None
              cleaned = value.lstrip("./")
              candidate = (pkg_dir / cleaned).resolve()
              if not candidate.exists():
                  return None
              try:
                  return candidate.relative_to(repo_root).as_posix()
              except ValueError:
                  return None

          obj,

          key_path: str,
          condition: str | None,
          callback,
          source: str,
      ) -> None:
          """Recursively walk a package.json exports object, calling callback
          for each string-valued leaf with a derived name."""
          if isinstance(obj, str):
              is_wildcard = "*" in obj
              callback(
                  key_path,
                  None if is_wildcard else obj,
                  entry_kind="library",
                  source=source,
                  condition=condition,
                  is_wildcard=is_wildcard,
                  path_pattern=obj if is_wildcard else None,
              )
              return
          if isinstance(obj, dict):
